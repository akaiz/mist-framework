46       private  void downloadUsingNIO(String urlStr, String file) throws IOException { 
47           URL url = new URL(urlStr); 
48           ReadableByteChannel rbc = Channels.newChannel(url.openStream()); 
49           FileOutputStream fos = new FileOutputStream(file); 
50           fos.getChannel().transferFrom(rbc, 0, Long.MAX_VALUE); 
51           fos.close(); 
52           rbc.close(); 
53       }
