81               for (int y = 0; y < image.getHeight(); y++) { 
82                   final int clr = image.getRGB(x, y); 
83                   final int red = (clr & 0x00ff0000) >> 16; 
84                   final int green = (clr & 0x0000ff00) >> 8; 
85                   final int blue = clr & 0x000000ff; 
86                   if (red == 255) { 
87                       System.out.println(String.format("Coordinate %d %d", x, y)); 
88                       return String.format("Coordinate %d %d", x, y); 
89                   } else { 
90                       return red+"--"+green+"--"+blue; 
91                   } 
92               } 
93           }

185      private static void unzipFunction(String destinationFolder, String zipFile) { 
186          File directory = new File(destinationFolder); 
187   
188          // if the output directory doesn't exist, create it 
189          if(!directory.exists()) 
190              directory.mkdirs(); 
191   
192          // buffer for read and write data to file 
193          byte[] buffer = new byte[2048]; 
194   
195          try { 
196              FileInputStream fInput = new FileInputStream(zipFile); 
197              ZipInputStream zipInput = new ZipInputStream(fInput); 
198   
199              ZipEntry entry = zipInput.getNextEntry(); 
200   
201              while(entry != null){ 
202                  String entryName = entry.getName(); 
203                  File file = new File(destinationFolder + File.separator + entryName); 
204   
205                  System.out.println("Unzip file " + entryName + " to " + file.getAbsolutePath()); 
206   
207                  // create the directories of the zip directory 
208                  if(entry.isDirectory()) { 
209                      File newDir = new File(file.getAbsolutePath()); 
210                      if(!newDir.exists()) { 
211                          boolean success = newDir.mkdirs(); 
212                          if(success == false) { 
213                              System.out.println("Problem creating Folder"); 
214                          } 
215                      } 
216                  } 
217                  else { 
218                      FileOutputStream fOutput = new FileOutputStream(file); 
219                      int count = 0; 
220                      while ((count = zipInput.read(buffer)) > 0) { 
221                          // write 'count' bytes to the file output stream 
222                          fOutput.write(buffer, 0, count); 
223                      } 
224                      fOutput.close(); 
225                  } 
226                  // close ZipEntry and take the next one 
227                  zipInput.closeEntry(); 
228                  entry = zipInput.getNextEntry(); 
229              } 
230   
231              // close the last ZipEntry 
232              zipInput.closeEntry(); 
233   
234              zipInput.close(); 
235              fInput.close(); 
236   
237              File zip = new File(zipFile); 
238              if(zip.exists()){ 
239                  zip.delete(); 
240              } 
241   
242              File temp = new File(destinationFolder+"/__MACOSX"); 
243              if(temp.exists()){ 
244                  temp.delete(); 
245              } 
246          } catch (IOException e) { 
247              e.printStackTrace(); 
248          } 
249      }

253      { 
254          URL url = new URL(sourceURL); 
255          String fileName = sourceURL.substring(sourceURL.lastIndexOf('/') + 1, sourceURL.length()); 
256          Path targetPath = new File(targetDirectory + File.separator + fileName).toPath(); 
257          Files.copy(url.openStream(), targetPath, StandardCopyOption.REPLACE_EXISTING); 
258

261      private  void downloadUsingNIO(String urlStr, String file) throws IOException { 
262          URL url = new URL(urlStr); 
263          ReadableByteChannel rbc = Channels.newChannel(url.openStream()); 
264          FileOutputStream fos = new FileOutputStream(file); 
265          fos.getChannel().transferFrom(rbc, 0, Long.MAX_VALUE); 
266          fos.close(); 
267          rbc.close(); 
268      }

295      public static String getMostCommonColour(Map map) { 
296          List list = new LinkedList(map.entrySet()); 
297          Collections.sort(list, new Comparator() { 
298              public int compare(Object o1, Object o2) { 
299                  return ((Comparable) ((Map.Entry) (o1)).getValue()) 
300                          .compareTo(((Map.Entry) (o2)).getValue()); 
301              } 
302          }); 
303          Map.Entry me = (Map.Entry )list.get(list.size()-1); 
304          int[] rgb= getRGBArr((Integer)me.getKey()); 
305          return Integer.toHexString(rgb[0])+""+Integer.toHexString(rgb[1])+""+ 
306                  Integer.toHexString(rgb[2]); 
307      }

309      public static int[] getRGBArr(int pixel) { 
310          int alpha = (pixel >> 24) & 0xff; 
311          int red = (pixel >> 16) & 0xff; 
312          int green = (pixel >> 8) & 0xff; 
313          int blue = (pixel) & 0xff; 
314          return new int[]{red,green,blue};

317      public static boolean isGray(int[] rgbArr) { 
318          int rgDiff = rgbArr[0] - rgbArr[1]; 
319          int rbDiff = rgbArr[0] - rgbArr[2]; 
320          // Filter out black, white and grays...... (tolerance within 10 pixels) 
321          int tolerance = 10; 
322          if (rgDiff > tolerance || rgDiff < -tolerance) 
323              if (rbDiff > tolerance || rbDiff < -tolerance) { 
324                  return false; 
325              } 
326          return true; 
327      }
