533      public  void deployNodsThread(Node node1,Node node2,CredentialsProvider credsProvider) 
534      { 
535   
536          //create a callable for each method 
537          Callable<Void> callable1 = new Callable<Void>() 
538          { 
539              @Override 
540              public Void call() throws Exception 
541              { 
542                  System.out.println("Node 1"); 
543                  deploy(node1, credsProvider, 1); 
544                  return null; 
545              } 
546          }; 
547   
548          Callable<Void> callable2 = new Callable<Void>() 
549          { 
550              @Override 
551              public Void call() throws Exception 
552              { 
553                  System.out.println("Node 2"); 
554                  deploy(node2, credsProvider, 2); 
555                  return null; 
556              } 
557          }; 
558   
559   
560          //add to a list 
561          List<Callable<Void>> taskList = new ArrayList<Callable<Void>>(); 
562          taskList.add(callable1); 
563          taskList.add(callable2); 
564   
565   
566          //create a pool executor with 3 threads 
567          ExecutorService executor = Executors.newFixedThreadPool(2); 
568   
569          try 
570          { 
571              //start the threads 
572              List<Future<Void>> futureList = executor.invokeAll(taskList); 
573   
574              for(Future<Void> voidFuture : futureList) 
575              { 
576                  try 
577                  { 
578                      //check the status of each future.  get will block until the task 
579                      //completes or the time expires 
580                      voidFuture.get(100000, TimeUnit.MILLISECONDS); 
581                  } 
582                  catch (ExecutionException e) 
583                  { 
584                      System.err.println("Error executing task " + e.getMessage()); 
585                  } 
586                  catch (TimeoutException e) 
587                  { 
588                      System.err.println("Timed out executing task" + e.getMessage()); 
589                  } 
590   
591              } 
592   
593   
594          } 
595          catch (InterruptedException ie) 
596          { 
597              //do something if you care about interruption; 
598          } 
599   
600      }

971      public String executeRequest(HttpRequestBase requestBase, CredentialsProvider credsProvider) throws ClientProtocolException { 
972          CloseableHttpClient client = HttpClients.custom().setDefaultCredentialsProvider(credsProvider).build(); 
973          InputStream responseStream = null; 
974          String res = null; 
975          HttpResponse response = null; 
976          try { 
977              response = client.execute(requestBase); 
978          } catch (IOException e) { 
979              System.out.println("Retry sending request"); 
980             executeRequest(requestBase,credsProvider); 
981          } 
982          HttpEntity responseEntity = response.getEntity() ; 
983          try { 
984              responseStream = responseEntity.getContent() ; 
985          } catch (IOException e) { 
986              e.printStackTrace(); 
987          } 
988   
989          BufferedReader br = new BufferedReader (new InputStreamReader (responseStream)) ; 
990          StringBuilder sb = new StringBuilder(); 
991          String line; 
992          try { 
993              while ((line = br.readLine()) != null) { 
994                  sb.append(line); 
995                  sb.append(System.getProperty("line.separator")); 
996              } 
997          } catch (IOException e) { 
998              e.printStackTrace(); 
999          } 
1000         try { 
1001             br.close() ; 
1002         } catch (IOException e) { 
1003             e.printStackTrace(); 
1004         } 
1005         res = sb.toString(); 
1006  
1007         return res;

1063     public static String getIpAddress() throws IOException { 
1064         String ip=null; 
1065         try { 
1066             Enumeration<NetworkInterface> interfaces = NetworkInterface.getNetworkInterfaces(); 
1067             while (interfaces.hasMoreElements()) { 
1068                 NetworkInterface iface = interfaces.nextElement(); 
1069                 // filters out 127.0.0.1 and inactive interfaces 
1070                 if (iface.isLoopback() || !iface.isUp()) 
1071                     continue; 
1072  
1073                 Enumeration<InetAddress> addresses = iface.getInetAddresses(); 
1074                 while(addresses.hasMoreElements()) { 
1075                     InetAddress addr = addresses.nextElement(); 
1076  
1077                     // *EDIT* 
1078                     if (addr instanceof Inet6Address) continue; 
1079  
1080                     ip = addr.getHostAddress(); 
1081                     System.out.println(iface.getDisplayName() + " " + ip); 
1082                 } 
1083             } 
1084         } catch (SocketException e) { 
1085             throw new RuntimeException(e); 
1086         } 
1087         return ip; 
1088     }
